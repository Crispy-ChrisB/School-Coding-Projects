        C:\xampp\htdocs\csc\project01b.php
        C:\xampp\htdocs\csc\project01c.php
        ...

        C:\xampp\htdocs\img\Chinook_Project\Album00002.jpg
        C:\xampp\htdocs\img\Chinook_Project\Album00003.jpg
        ...
